224 //*****************************************************************************
225 // Reset entry point for your code.
226 // Sets up a simple runtime environment and initializes the C/C++
227 // library.
228 //
229 //*****************************************************************************
230 void
231 ResetISR(void) {
232  unsigned long *pulSrc, *pulDest;
233 
234  //
235  // Copy the data segment initializers from flash to SRAM.
236  //
237  pulSrc = &_etext;
238  for(pulDest = &_data; pulDest < &_edata; )
239  {
240  *pulDest++ = *pulSrc++;
241  }
242 
243  //
244  // Zero fill the bss segment. This is done with inline assembly since this
245  // will clear the value of pulDest if it is not kept in a register.
246  //
247  __asm(" ldr r0, =_bss\n"
248  " ldr r1, =_ebss\n"
249  " mov r2, #0\n"
250  " .thumb_func\n"
251  "zero_loop:\n"
252  " cmp r0, r1\n"
253  " it lt\n"
254  " strlt r2, [r0], #4\n"
255  " blt zero_loop");
256 
257 #ifdef __USE_CMSIS
258  SystemInit();
259 #endif
260 
261 #if defined (__cplusplus)
262  //
263  // Call C++ library initialisation
264  //
265  __libc_init_array();
266 #endif
267 
268 #if defined (__REDLIB__)
269  // Call the Redlib library, which in turn calls main()
270  __main() ;
271 #else
272  main();
273 #endif
274 
275  //
276  // main() shouldn't return, but if it does, we'll just enter an infinite loop
277  //
278  while (1) {
279  ;
280  }
281 }
