127 portSTACK_TYPE *pxPortInitialiseStack( portSTACK_TYPE *pxTopOfStack, pdTASK_CODE pxCode, void *pvParameters )
128 {
129  /* Simulate the stack frame as it would be created by a context switch
130  interrupt. */
131  pxTopOfStack--; /* Offset added to account for the way the MCU uses the stack on entry/exit of interrupts. */
132  *pxTopOfStack = portINITIAL_XPSR; /* xPSR */
133  pxTopOfStack--;
134  *pxTopOfStack = ( portSTACK_TYPE ) pxCode; /* PC */
135  pxTopOfStack--;
136  *pxTopOfStack = 0; /* LR */
137  pxTopOfStack -= 5; /* R12, R3, R2 and R1. */
138  *pxTopOfStack = ( portSTACK_TYPE ) pvParameters; /* R0 */
139  pxTopOfStack -= 8; /* R11, R10, R9, R8, R7, R6, R5 and R4. */
140 
141  return pxTopOfStack;
142 }

145 void vPortSVCHandler( void )
146 {
147  __asm volatile (
148  " ldr r3, pxCurrentTCBConst2 \n" /* Restore the context. */
149  " ldr r1, [r3] \n" /* Use pxCurrentTCBConst to get the pxCurrentTCB address. */
150  " ldr r0, [r1] \n" /* The first item in pxCurrentTCB is the task top of stack. */
151  " ldmia r0!, {r4-r11} \n" /* Pop the registers that are not automatically saved on exception entry and the critical nesting count. */
152  " msr psp, r0 \n" /* Restore the task stack pointer. */
153  " mov r0, #0 \n"
154  " msr basepri, r0 \n"
155  " orr r14, #0xd \n"
156  " bx r14 \n"
157  " \n"
158  " .align 2 \n"
159  "pxCurrentTCBConst2: .word pxCurrentTCB \n"
160  );
161 }

164 static void prvPortStartFirstTask( void )
165 {
166  __asm volatile(
167  " ldr r0, =0xE000ED08 \n" /* Use the NVIC offset register to locate the stack. */
168  " ldr r0, [r0] \n"
169  " ldr r0, [r0] \n"
170  " msr msp, r0 \n" /* Set the msp back to the start of the stack. */
171  " cpsie i \n" /* Globally enable interrupts. */
172  " svc 0 \n" /* System call to start first task. */
173  " nop \n"
174  );
175 }

181 portBASE_TYPE xPortStartScheduler( void )
182 {
183  /* configMAX_SYSCALL_INTERRUPT_PRIORITY must not be set to 0.
184  See http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html */
185  configASSERT( configMAX_SYSCALL_INTERRUPT_PRIORITY );
186 
187  /* Make PendSV, CallSV and SysTick the same priroity as the kernel. */
188  *(portNVIC_SYSPRI2) |= portNVIC_PENDSV_PRI;
189  *(portNVIC_SYSPRI2) |= portNVIC_SYSTICK_PRI;
190 
191  /* Start the timer that generates the tick ISR. Interrupts are disabled
192  here already. */
193  prvSetupTimerInterrupt();
194 
195  /* Initialise the critical nesting count ready for the first task. */
196  uxCriticalNesting = 0;
197 
198  /* Start the first task. */
199  prvPortStartFirstTask();
200 
201  /* Should not get here! */
202  return 0;
203 }

237 void xPortPendSVHandler( void )
238 {
239  /* This is a naked function. */
240 
241  __asm volatile
242  (
243  " mrs r0, psp \n"
244  " \n"
245  " ldr r3, pxCurrentTCBConst \n" /* Get the location of the current TCB. */
246  " ldr r2, [r3] \n"
247  " \n"
248  " stmdb r0!, {r4-r11} \n" /* Save the remaining registers. */
249  " str r0, [r2] \n" /* Save the new top of stack into the first member of the TCB. */
250  " \n"
251  " stmdb sp!, {r3, r14} \n"
252  " mov r0, %0 \n"
253  " msr basepri, r0 \n"
254  " bl vTaskSwitchContext \n"
255  " mov r0, #0 \n"
256  " msr basepri, r0 \n"
257  " ldmia sp!, {r3, r14} \n"
258  " \n" /* Restore the context, including the critical nesting count. */
259  " ldr r1, [r3] \n"
260  " ldr r0, [r1] \n" /* The first item in pxCurrentTCB is the task top of stack. */
261  " ldmia r0!, {r4-r11} \n" /* Pop the registers. */
262  " msr psp, r0 \n"
263  " bx r14 \n"
264  " \n"
265  " .align 2 \n"
266  "pxCurrentTCBConst: .word pxCurrentTCB \n"
267  ::"i"(configMAX_SYSCALL_INTERRUPT_PRIORITY)
268  );
269 }

272 void xPortSysTickHandler( void )
273 {
274 unsigned long ulDummy;
275 
276  /* If using preemption, also force a context switch. */
277  #if configUSE_PREEMPTION == 1
278  *(portNVIC_INT_CTRL) = portNVIC_PENDSVSET;
279  #endif
280 
281  ulDummy = portSET_INTERRUPT_MASK_FROM_ISR();
282  {
283  vTaskIncrementTick();
284  }
285  portCLEAR_INTERRUPT_MASK_FROM_ISR( ulDummy );
286 }

293 void prvSetupTimerInterrupt( void )
294 {
295  /* Configure SysTick to interrupt at the requested rate. */
296  *(portNVIC_SYSTICK_LOAD) = ( configCPU_CLOCK_HZ / configTICK_RATE_HZ ) - 1UL;
