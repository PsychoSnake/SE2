121 /*
122  * Set basepri to portMAX_SYSCALL_INTERRUPT_PRIORITY without effecting other
123  * registers. r0 is clobbered.
124  */
125 #define portSET_INTERRUPT_MASK() \
126  __asm volatile \
127  ( \
128  " mov r0, %0 \n" \
129  " msr basepri, r0 \n" \
130  ::"i"(configMAX_SYSCALL_INTERRUPT_PRIORITY):"r0" \
131  )
132 
133 /*
134  * Set basepri back to 0 without effective other registers.
135  * r0 is clobbered. FAQ: Setting BASEPRI to 0 is not a bug. Please see
136  * http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html before disagreeing.
137  */
138 #define portCLEAR_INTERRUPT_MASK() \
139  __asm volatile \
140  ( \
141  " mov r0, #0 \n" \
142  " msr basepri, r0 \n" \
143  :::"r0" \
144  )
145 
146 /* FAQ: Setting BASEPRI to 0 in portCLEAR_INTERRUPT_MASK_FROM_ISR() is not a
147 bug. Please see http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html before
148 disagreeing. */
149 #define portSET_INTERRUPT_MASK_FROM_ISR() 0;portSET_INTERRUPT_MASK()
