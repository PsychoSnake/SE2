139 unsigned char
140 webclient_get(char *host, u16_t port, char *file)
141 {
142  struct uip_conn *conn;
143  uip_ipaddr_t *ipaddr;
144  static uip_ipaddr_t addr;
145 
146  /* First check if the host is an IP address. */
147  ipaddr = &addr;
148  if(uiplib_ipaddrconv(host, (unsigned char *)addr) == 0) {
149  ipaddr = (uip_ipaddr_t *)resolv_lookup(host);
150 
151  if(ipaddr == NULL) {
152  return 0;
153  }
154  }
155 
156  conn = uip_connect(ipaddr, htons(port));
157 
158  if(conn == NULL) {
159  return 0;
160  }
161 
162  s.port = port;
163  strncpy(s.file, file, sizeof(s.file));
164  strncpy(s.host, host, sizeof(s.host));
165 
166  init_connection();
167  return 1;
168 }

178 static void
179 senddata(void)
180 {
181  u16_t len;
182  char *getrequest;
183  char *cptr;
184 
185  if(s.getrequestleft > 0) {
186  cptr = getrequest = (char *)uip_appdata;
187 
188  cptr = copy_string(cptr, http_get, sizeof(http_get) - 1);
189  cptr = copy_string(cptr, s.file, strlen(s.file));
190  *cptr++ = ISO_space;
191  cptr = copy_string(cptr, http_10, sizeof(http_10) - 1);
192 
193  cptr = copy_string(cptr, http_crnl, sizeof(http_crnl) - 1);
194 
195  cptr = copy_string(cptr, http_host, sizeof(http_host) - 1);
196  cptr = copy_string(cptr, s.host, strlen(s.host));
197  cptr = copy_string(cptr, http_crnl, sizeof(http_crnl) - 1);
198 
199  cptr = copy_string(cptr, http_user_agent_fields,
200  strlen(http_user_agent_fields));
201 
202  len = s.getrequestleft > uip_mss()?
203  uip_mss():
204  s.getrequestleft;
205  uip_send(&(getrequest[s.getrequestptr]), len);
206  }
207 }

223 static u16_t
224 parse_statusline(u16_t len)
225 {
226  char *cptr;
227 
228  while(len > 0 && s.httpheaderlineptr < sizeof(s.httpheaderline)) {
229  s.httpheaderline[s.httpheaderlineptr] = *(char *)uip_appdata;
230  char * aux = (char *)uip_appdata;
231  uip_appdata = aux + 1;
232  --len;
233  if(s.httpheaderline[s.httpheaderlineptr] == ISO_nl) {
234 
235  if((strncmp(s.httpheaderline, http_10,
236  sizeof(http_10) - 1) == 0) ||
237  (strncmp(s.httpheaderline, http_11,
238  sizeof(http_11) - 1) == 0)) {
239  cptr = &(s.httpheaderline[9]);
240  s.httpflag = HTTPFLAG_NONE;
241  if(strncmp(cptr, http_200, sizeof(http_200) - 1) == 0) {
242  /* 200 OK */
243  s.httpflag = HTTPFLAG_OK;
244  } else if(strncmp(cptr, http_301, sizeof(http_301) - 1) == 0 ||
245  strncmp(cptr, http_302, sizeof(http_302) - 1) == 0) {
246  /* 301 Moved permanently or 302 Found. Location: header line
247  will contain thw new location. */
248  s.httpflag = HTTPFLAG_MOVED;
249  } else {
250  s.httpheaderline[s.httpheaderlineptr - 1] = 0;
251  }
252  } else {
253  uip_abort();
254  webclient_aborted();
255  return 0;
256  }
257 
258  /* We're done parsing the status line, so we reset the pointer
259  and start parsing the HTTP headers.*/
260  s.httpheaderlineptr = 0;
261  s.state = WEBCLIENT_STATE_HEADERS;
262  break;
263  } else {
264  ++s.httpheaderlineptr;
265  }
266  }
267  return len;
268 }

270 static char
271 casecmp(char *str1, const char *str2, char len)
272 {
273  static char c;
274 
275  while(len > 0) {
276  c = *str1;
277  /* Force lower-case characters. */
278  if(c & 0x40) {
279  c |= 0x20;
280  }
281  if(*str2 != c) {
282  return 1;
283  }
284  ++str1;
285  ++str2;
286  --len;
287  }
288  return 0;
289 }

291 static u16_t
292 parse_headers(u16_t len)
293 {
294  char *cptr;
295  static unsigned char i;
296 
297  while(len > 0 && s.httpheaderlineptr < sizeof(s.httpheaderline)) {
298  s.httpheaderline[s.httpheaderlineptr] = *(char *)uip_appdata;
299  char * aux = (char *)uip_appdata;
300  uip_appdata = aux + 1;
301  --len;
302  if(s.httpheaderline[s.httpheaderlineptr] == ISO_nl) {
303  /* We have an entire HTTP header line in s.httpheaderline, so
304  we parse it. */
305  if(s.httpheaderline[0] == ISO_cr) {
306  /* This was the last header line (i.e., and empty "\r\n"), so
307  we are done with the headers and proceed with the actual
308  data. */
309  s.state = WEBCLIENT_STATE_DATA;
310  return len;
311  }
312 
313  s.httpheaderline[s.httpheaderlineptr - 1] = 0;
314  /* Check for specific HTTP header fields. */
315  if(casecmp(s.httpheaderline, http_content_type,
316  sizeof(http_content_type) - 1) == 0) {
317  /* Found Content-type field. */
318  cptr = strchr(s.httpheaderline, ';');
319  if(cptr != NULL) {
320  *cptr = 0;
321  }
322  strncpy(s.mimetype, s.httpheaderline +
323  sizeof(http_content_type) - 1, sizeof(s.mimetype));
324  } else if(casecmp(s.httpheaderline, http_location,
325  sizeof(http_location) - 1) == 0) {
326  cptr = s.httpheaderline +
327  sizeof(http_location) - 1;
328 
329  if(strncmp(cptr, http_http, 7) == 0) {
330  cptr += 7;
331  for(i = 0; i < s.httpheaderlineptr - 7; ++i) {
332  if(*cptr == 0 ||
333  *cptr == '/' ||
334  *cptr == ' ' ||
335  *cptr == ':') {
336  s.host[i] = 0;
337  break;
338  }
339  s.host[i] = *cptr;
340  ++cptr;
341  }
342  }
343  strncpy(s.file, cptr, sizeof(s.file));
344  /* s.file[s.httpheaderlineptr - i] = 0;*/
345  }
346 
347 
348  /* We're done parsing, so we reset the pointer and start the
349  next line. */
350  s.httpheaderlineptr = 0;
351  } else {
352  ++s.httpheaderlineptr;
353  }
354  }
355  return len;
356 }

358 static void
359 newdata(void)
360 {
361  u16_t len;
362 
363  len = uip_datalen();
364 
365  if(s.state == WEBCLIENT_STATE_STATUSLINE) {
366  len = parse_statusline(len);
367  }
368 
369  if(s.state == WEBCLIENT_STATE_HEADERS && len > 0) {
370  len = parse_headers(len);
371  }
372 
373  if(len > 0 && s.state == WEBCLIENT_STATE_DATA &&
374  s.httpflag != HTTPFLAG_MOVED) {
375  webclient_datahandler((char *)uip_appdata, len);
376  }
377 }

379 void
380 webclient_appcall(void)
381 {
382  if(uip_connected()) {
383  s.timer = 0;
384  s.state = WEBCLIENT_STATE_STATUSLINE;
385  senddata();
386  webclient_connected();
387  return;
388  }
389 
390  if(s.state == WEBCLIENT_STATE_CLOSE) {
391  webclient_closed();
392  uip_abort();
393  return;
394  }
395 
396  if(uip_aborted()) {
397  webclient_aborted();
398  }
399  if(uip_timedout()) {
400  webclient_timedout();
401  }
402 
403 
404  if(uip_acked()) {
405  s.timer = 0;
406  acked();
407  }
408  if(uip_newdata()) {
409  s.timer = 0;
410  newdata();
411  }
412  if(uip_rexmit() ||
413  uip_newdata() ||
414  uip_acked()) {
415  senddata();
416  } else if(uip_poll()) {
417  ++s.timer;
418  if(s.timer == WEBCLIENT_TIMEOUT) {
419  webclient_timedout();
420  uip_abort();
421  return;
422  }
423  /* senddata();*/
424  }
425 
426  if(uip_closed()) {
427  if(s.httpflag != HTTPFLAG_MOVED) {
428  /* Send NULL data to signal EOF. */
429  webclient_datahandler(NULL, 0);
430  } else {
431  if(resolv_lookup(s.host) == NULL) {
432  resolv_query(s.host);
433  }
434  webclient_get(s.host, s.port, s.file);
435  }
436  }
